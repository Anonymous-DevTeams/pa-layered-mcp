    Windows Installation Guide#

    Set up the PA-Layer environment on Windows 10/11.

    1. Install with Scoop (Recommended)#

     1
 2
 3
 4
 5
 6
 7

 9
10

    # Install Scoop (if not installed)
Set-ExecutionPolicy RemoteSigned -Scope CurrentUser
irm get.scoop.sh | iex

# Install PA-Retriever
scoop bucket add pa-layer https://github.com/pa-layer/scoop-bucket
scoop install pa-retriever

# Verify installation
pa version

    Download with PowerShell#

    # Navigate to Downloads
cd $HOME\Downloads

# Download
Invoke-WebRequest -Uri "https://github.com/pa-layer/pa-retriever/releases/latest/download/pa-retriever-windows-amd64.zip" -OutFile "pa-retriever.zip"

# Extract
Expand-Archive -Path "pa-retriever.zip" -DestinationPath "pa-retriever"

# Create install directory
New-Item -ItemType Directory -Force -Path "$HOME\.local\bin"

# Move binary
Move-Item -Path "pa-retriever\pa.exe" -Destination "$HOME\.local\bin\pa.exe"
