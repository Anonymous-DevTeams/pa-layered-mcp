    # ~/.pa/config.yaml

# Active profile
active: default

# Workspace settings
workspace:
  root: ~/playground
  obsidian_vault: ~/playground/mcp-docs
  code_space: ~/playground/CodeSpace

# Index settings
index:
  db_path: ~/.pa/default.db
  wal_mode: true
  exclude:
    - "*.tmp"
    - ".git"
    - "node_modules"
    - ".obsidian/workspace*.json"

# Document type settings
types:
  preset: agile  # basic, agile, design, full

# Profile list
profiles:
  - name: default
    db_path: ~/.pa/default.db
    docs_path: ~/playground/mcp-docs
    description: "Default profile"

# MCP server settings
mcp:
  servers:
    - name: claude-desktop
      enabled: true
    - name: claude-code
      enabled: true
