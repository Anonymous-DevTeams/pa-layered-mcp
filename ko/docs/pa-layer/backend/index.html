    // application/app-api/src/.../adapter/LoginAdapter.kt
package com.example.application.adapter.auth

@Component
class LoginAdapter(
    private val userPort: UserPort,
    private val tokenPort: TokenPort,
    private val sessionPort: SessionPort,
    private val eventPublisher: EventPublisher
) : LoginPort {
    
    @Transactional
    override suspend fun login(request: LoginRequest): LoginResult {
        // 1. 사용자 조회 및 검증
        val user = userPort.findByEmail(request.email)
            ?: return LoginResult.InvalidCredentials
        
        // 2. 비밀번호 검증
        if (!user.verifyPassword(request.password)) {
            eventPublisher.publish(LoginFailedEvent(user.id))
            return LoginResult.InvalidCredentials
        }
        
        // 3. 계정 상태 확인
        if (user.isLocked) {
            return LoginResult.AccountLocked
        }
        
        // 4. 토큰 생성
        val tokens = tokenPort.generate(user.id, request.deviceInfo)
        
        // 5. 세션 생성
        sessionPort.create(user.id, tokens.sessionId)
        
        // 6. 이벤트 발행
        eventPublisher.publish(LoginSuccessEvent(user.id))
        
        return LoginResult.Success(
            accessToken = tokens.accessToken,
            refreshToken = tokens.refreshToken,
            expiresIn = tokens.expiresIn
        )
    }
}

    // src/domain/auth/adapter/login.adapter.ts
import { Injectable, Inject } from '@nestjs/common';
import { LoginPort, LoginRequest, LoginResult } from '../port/login.port';
import { USER_PORT, UserPort } from '../../users/port/user.port';
import { TOKEN_PORT, TokenPort } from '../port/token.port';

@Injectable()
export class LoginAdapter implements LoginPort {
  constructor(
    @Inject(USER_PORT) private readonly userPort: UserPort,
    @Inject(TOKEN_PORT) private readonly tokenPort: TokenPort,
  ) {}

  async login(request: LoginRequest): Promise<LoginResult> {
    // 1. 사용자 조회
    const user = await this.userPort.findByEmail(request.email);
    if (!user) {
      return { type: 'invalidCredentials' };
    }

    // 2. 비밀번호 검증
    const isValid = await this.userPort.verifyPassword(
      user.id,
      request.password
    );
    if (!isValid) {
      return { type: 'invalidCredentials' };
    }

    // 3. 토큰 생성
    const tokens = await this.tokenPort.generate(user.id);

    return {
      type: 'success',
      accessToken: tokens.accessToken,
      refreshToken: tokens.refreshToken,
      expiresIn: tokens.expiresIn,
    };
  }
}

    // internal/domain/auth/adapter.go
package auth

import (
    "context"
    "my-api/internal/domain/users"
)

type LoginAdapter struct {
    userPort  users.UserPort
    tokenPort TokenPort
}

func NewLoginAdapter(userPort users.UserPort, tokenPort TokenPort) *LoginAdapter {
    return &LoginAdapter{
        userPort:  userPort,
        tokenPort: tokenPort,
    }
}

func (a *LoginAdapter) Login(ctx context.Context, req LoginRequest) (LoginResult, error) {
    // 1. Find user
    user, err := a.userPort.FindByEmail(ctx, req.Email)
    if err != nil {
        return LoginResult{Type: LoginInvalidCredentials}, nil
    }

    // 2. Verify password
    if !user.VerifyPassword(req.Password) {
        return LoginResult{Type: LoginInvalidCredentials}, nil
    }

    // 3. Generate tokens
    tokens, err := a.tokenPort.Generate(ctx, user.ID)
    if err != nil {
        return LoginResult{}, err
    }

    return LoginResult{
        Type:         LoginSuccess,
        AccessToken:  tokens.AccessToken,
        RefreshToken: tokens.RefreshToken,
        ExpiresIn:    tokens.ExpiresIn,
    }, nil
}

    // src/domain/auth/adapter.rs
use async_trait::async_trait;
use std::sync::Arc;

use super::port::LoginPort;
use super::types::*;
use crate::domain::users::port::UserPort;

pub struct LoginAdapter {
    user_port: Arc<dyn UserPort>,
    token_port: Arc<dyn TokenPort>,
}

impl LoginAdapter {
    pub fn new(
        user_port: Arc<dyn UserPort>,
        token_port: Arc<dyn TokenPort>,
    ) -> Self {
        Self { user_port, token_port }
    }
}

#[async_trait]
impl LoginPort for LoginAdapter {
    async fn login(&self, request: LoginRequest) -> Result<LoginResult, AuthError> {
        // 1. Find user
        let user = self.user_port
            .find_by_email(&request.email)
            .await
            .map_err(|e| AuthError::Internal(e.to_string()))?;
        
        let user = match user {
            Some(u) => u,
            None => return Ok(LoginResult::InvalidCredentials),
        };

        // 2. Verify password
        if !user.verify_password(&request.password) {
            return Ok(LoginResult::InvalidCredentials);
        }

        // 3. Generate tokens
        let tokens = self.token_port
            .generate(&user.id)
            .await
            .map_err(|e| AuthError::Internal(e.to_string()))?;

        Ok(LoginResult::Success {
            access_token: tokens.access_token,
            refresh_token: tokens.refresh_token,
            expires_in: tokens.expires_in,
        })
    }
}
