    Apple Silicon (M1/M2/M3) 및 Intel Mac에서 PA-Layer 환경을 구성합니다.

brew tap pa-layer/tap
brew install pa-retriever

# 설치 확인
pa version

# 압축 해제
tar xzf pa-retriever-darwin-arm64.tar.gz

# 실행 권한 부여
chmod +x pa

# PATH에 추가 (zsh 기준)
sudo mv pa /usr/local/bin/
# 또는 사용자 디렉토리에 설치
mkdir -p ~/.local/bin
mv pa ~/.local/bin/
echo 'export PATH="$HOME/.local/bin:$PATH"' >> ~/.zshrc
source ~/.zshrc

    # PA-Layer 환경 변수
export PA_HOME="$HOME/.pa"
export PA_WORKSPACE="$HOME/playground"

# 단축 명령어 (선택)
alias pas="pa search"
alias pai="pa index"
alias pac="pa code"

# 자동완성 (선택)
if command -v pa &> /dev/null; then
  eval "$(pa completion zsh)"
fi
