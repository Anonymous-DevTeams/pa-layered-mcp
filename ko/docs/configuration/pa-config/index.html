    # ~/.pa/config.yaml

# 활성 프로필
active: default

# 워크스페이스 설정
workspace:
  root: ~/playground
  obsidian_vault: ~/playground/mcp-docs
  code_space: ~/playground/CodeSpace

# 인덱스 설정
index:
  db_path: ~/.pa/default.db
  wal_mode: true
  exclude:
    - "*.tmp"
    - ".git"
    - "node_modules"
    - ".obsidian/workspace*.json"
    - "daily/*"

# 문서 타입 설정
types:
  preset: agile
  custom: []

# 프로필 목록
profiles:
  - name: default
    db_path: ~/.pa/default.db
    docs_path: ~/playground/mcp-docs
    description: "기본 프로필"

# MCP 서버 설정
mcp:
  servers:
    - name: claude-desktop
      enabled: true
    - name: claude-code
      enabled: true

    index:
  # SQLite 데이터베이스 경로
  db_path: ~/.pa/default.db
  
  # WAL 모드 (성능 향상, 권장)
  wal_mode: true
  
  # 인덱싱 제외 패턴
  exclude:
    # 임시 파일
    - "*.tmp"
    - "*.bak"
    
    # 버전 관리
    - ".git"
    - ".svn"
    
    # 의존성
    - "node_modules"
    - "vendor"
    - ".gradle"
    - "target"
    
    # Obsidian 임시 파일
    - ".obsidian/workspace*.json"
    - ".trash"
    
    # 사용자 정의
    - "daily/*"           # 일일 노트 제외
    - "attachments/*"     # 첨부파일 제외

    types:
  # 프리셋 선택
  # basic: l1, lm, l2, api, query, feature, template, skill, guide
  # agile: basic + epic, story, task, bug, spike, intake, backlog
  # design: basic + index, convention, adr, rfc, spec, diagram
  # full: 모든 타입
  preset: agile
  
  # 커스텀 타입 추가
  custom:
    - name: memo
      description: "짧은 메모"
    - name: meeting
      description: "회의록"

    # ~/playground/mcp-docs/.pa-index.yaml
name: my-project

# 이 Vault에서만 사용할 추가 타입
types:
  - custom-type

# 이 Vault의 제외 패턴 (전역에 추가)
exclude:
  - "drafts/*"
  - "archive/*"

# 타입 자동 추론 규칙
infer:
  - pattern: "**/L1-*.md"
    type: l1
  - pattern: "**/L2-*.md"
    type: l2
  - pattern: "**/LM-*.md"
    type: lm
  - pattern: "**/API-*.md"
    type: api
  - pattern: "**/Q-*.md"
    type: query
